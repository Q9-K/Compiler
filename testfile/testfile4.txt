// 算术运算
int main()
{
    printf("20374319\n");
    printf("%d\n", 1 + 2);
    printf("%d\n", 1 - 2);
    printf("%d\n", 1 / 2);
    printf("%d\n", 1 % 2);
    printf("%d\n", (1 + 2) + (1 - 2));
    printf("%d\n", (1 + 2) / (1 - 2));
    printf("%d\n", (1 + 2) * (1 - 2));
    printf("%d\n", (1 + 2) % (1 - 2));
    return 0;
}
